```tsx
/**
 * ConvergenceOverlay — Multi-series log-scale convergence chart (PIPE-12).
 *
 * Shows residual vs iteration curves for all selected cases on a single
 * Recharts LineChart with log-scale Y-axis.
 *
 * Pattern from ResidualChart.tsx + 34-UI-SPEC.md.
 */

import React, { useMemo } from 'react';
import {
  LineChart,
  Line,
  XAxis,
  YAxis,
  CartesianGrid,
  Tooltip,
  Legend,
  ResponsiveContainer,
} from 'recharts';
import type { ConvergenceDataPoint } from '../services/types';

const SERIES_COLORS = [
  '#ef4444', '#22c55e', '#3b82f6', '#a855f7',
  '#f59e0b', '#06b6d4', '#ec4899', '#84cc16',
];

interface SeriesDef {
  caseId: string;
  label: string;        // e.g. "CASE-XXXXXXXX (velocity=1)"
  color: string;
  data: ConvergenceDataPoint[];
}

interface ConvergenceOverlayProps {
  series: SeriesDef[];
  referenceCaseId?: string;
}

export default function ConvergenceOverlay({ series, referenceCaseId }: ConvergenceOverlayProps) {
  // Merge all series into iteration-keyed data for Recharts
  const merged = useMemo(() => {
    const iterationSet = new Set<number>();
    series.forEach(s => s.data.forEach(d => iterationSet.add(d.iteration)));
    return Array.from(iterationSet)
      .sort((a, b) => a - b)
      .map(iter => {
        const point: Record<string, number> = { iteration: iter };
        series.forEach(s => {
          const dp = s.data.find(d => d.iteration === iter);
          // Use p residual if available, else Ux
          point[s.caseId] = dp?.p ?? dp?.Ux ?? NaN;
        });
        return point;
      });
  }, [series]);

  // Compute domain for Y axis (log scale needs positive values)
  const yDomain = useMemo(() => {
    const allValues = merged.flatMap(p =>
      series.map(s => p[s.caseId]).filter(v => !isNaN(v) && v > 0)
    );
    if (allValues.length === 0) return [1e-10, 1];
    const min = Math.min(...allValues);
    const max = Math.max(...allValues);
    return [min * 0.5, max * 2];
  }, [merged, series]);

  if (series.length === 0 || merged.length === 0) {
    return (
      <div style={{ textAlign: 'center', padding: '48px', color: 'var(--text-secondary, #666)' }}>
        No convergence data available.
      </div>
    );
  }

  const CustomTooltip = ({ active, payload, label }: any) => {
    if (!active || !payload?.length) return null;
    return (
      <div style={{ background: '#fff', border: '1px solid #ddd', padding: '8px', fontSize: '12px' }}>
        <p><strong>Iteration {label}</strong></p>
        {payload.map((entry: any) => (
          <p key={entry.dataKey} style={{ color: entry.color }}>
            {series.find(s => s.caseId === entry.dataKey)?.label}: {entry.value?.toExponential(4)}
          </p>
        ))}
      </div>
    );
  };

  return (
    <ResponsiveContainer width="100%" height={400}>
      <LineChart data={merged} margin={{ top: 20, right: 30, left: 20, bottom: 20 }}>
        <CartesianGrid strokeDasharray="3 3" />
        <XAxis dataKey="iteration" label={{ value: 'Iteration', position: 'bottom', offset: 0 }} />
        <YAxis
          type="number"
          scale="log"
          domain={yDomain}
          tickFormatter={(v: number) => v.toExponential(0)}
          width={80}
        />
        <Tooltip content={<CustomTooltip />} />
        <Legend
          formatter={(value: string) => {
            const s = series.find(s => s.caseId === value);
            return s ? s.label : value;
          }}
        />
        {series.map((s, i) => (
          <Line
            key={s.caseId}
            type="monotone"
            dataKey={s.caseId}
            stroke={s.color}
            dot={false}
            connectNulls={false}
            isAnimationActive={false}
            strokeWidth={1.5}
          />
        ))}
      </LineChart>
    </ResponsiveContainer>
  );
}
```

```tsx
/**
 * DeltaFieldViewer — Displays computed delta field VTU in an iframe (PIPE-12).
 *
 * Calls POST /comparisons/{id}/delta-session which launches a trame session
 * on the delta .vtu file and returns {trame_url} for iframe src.
 * Shows loading spinner during computation and error state on failure.
 *
 * Pattern from 34-UI-SPEC.md + trame iframe pattern.
 */

import React, { useState, useCallback } from 'react';
import { api } from '../services/api';

interface DeltaFieldViewerProps {
  comparisonId: string;
  deltaVtuUrl?: string;       // URL to delta VTU file from comparison response
  deltaFieldName: string;
  caseAId: string;
  caseBId: string;
  meshHashMismatch?: boolean;  // true if mesh_seed_hash differs between cases
}

export default function DeltaFieldViewer({
  comparisonId,
  deltaVtuUrl,
  deltaFieldName,
  caseAId,
  caseBId,
  meshHashMismatch,
}: DeltaFieldViewerProps) {
  const [loading, setLoading] = useState(false);
  const [error, setError] = useState<string | null>(null);
  const [trameUrl, setTrameUrl] = useState<string | null>(deltaVtuUrl ?? null);

  const handleCompute = useCallback(async () => {
    setLoading(true);
    setError(null);
    try {
      // Call POST /comparisons/{id}/delta-session → returns {trame_url}
      const { trame_url } = await api.launchDeltaSession(comparisonId, deltaFieldName);
      if (trame_url) {
        setTrameUrl(trame_url);
      } else {
        setError('Delta session returned no trame URL.');
      }
    } catch (e: any) {
      setError(`Delta computation failed: ${e.message}`);
    } finally {
      setLoading(false);
    }
  }, [comparisonId, deltaFieldName]);

  if (meshHashMismatch) {
    return (
      <div style={{
        padding: '16px',
        background: 'rgba(245, 158, 11, 0.1)',
        border: '1px solid #f59e0b',
        borderRadius: '4px',
        marginBottom: '16px',
      }}>
        <strong style={{ color: '#92400e' }}>Delta field requires identical mesh topology.</strong>
        {' '}Cases with different mesh seed hashes may produce invalid results.
      </div>
    );
  }

  if (loading) {
    return (
      <div style={{ display: 'flex', alignItems: 'center', justifyContent: 'center', height: '400px' }}>
        <div style={{ textAlign: 'center' }}>
          <div style={{ fontSize: '24px', marginBottom: '8px' }}>Computing delta field...</div>
          <div style={{ color: '#666' }}>{caseBId} - {caseAId} ({deltaFieldName})</div>
        </div>
      </div>
    );
  }

  if (error) {
    return (
      <div style={{ padding: '16px', textAlign: 'center' }}>
        <div style={{ color: '#ef4444', marginBottom: '12px' }}>{error}</div>
        <button onClick={handleCompute}>Retry</button>
      </div>
    );
  }

  if (!trameUrl) {
    return (
      <div style={{ padding: '32px', textAlign: 'center' }}>
        <div style={{ marginBottom: '16px', color: '#666' }}>
          No delta field computed yet.
        </div>
        <button onClick={handleCompute}>
          Compute Delta Field
        </button>
      </div>
    );
  }

  return (
    <iframe
      src={trameUrl}
      style={{ width: '100%', height: '600px', border: '1px solid #ddd', borderRadius: '4px' }}
      title={`Delta Field: ${caseBId} - ${caseAId}`}
    />
  );
}
```

```tsx
/**
 * MetricsTable — Sortable metrics table with CSV export (PIPE-12).
 *
 * Columns: Case ID | Params | Final Residual | Execution Time | Diff %
 * Sortable by any column; Diff % highlighted amber if |diff| > 10%.
 * Export produces a CSV blob download.
 *
 * Pattern from SweepDetailPage.tsx + 34-UI-SPEC.md.
 */

import React, { useState, useMemo } from 'react';
import type { MetricsRow, ComparisonResponse } from '../services/types';

type SortKey = keyof MetricsRow;

interface MetricsTableProps {
  rows: MetricsRow[];
  referenceCaseId: string;
}

export default function MetricsTable({ rows, referenceCaseId }: MetricsTableProps) {
  const [sortKey, setSortKey] = useState<SortKey>('final_residual');
  const [sortAsc, setSortAsc] = useState<boolean>(true);

  const sorted = useMemo(() => {
    return [...rows].sort((a, b) => {
      const av = a[sortKey] ?? (sortKey === 'params' ? '' : Infinity);
      const bv = b[sortKey] ?? (sortKey === 'params' ? '' : Infinity);
      if (av === bv) return 0;
      const cmp = av < bv ? -1 : 1;
      return sortAsc ? cmp : -cmp;
    });
  }, [rows, sortKey, sortAsc]);

  const handleSort = (key: SortKey) => {
    if (sortKey === key) {
      setSortAsc(a => !a);
    } else {
      setSortKey(key);
      setSortAsc(true);
    }
  };

  const downloadCSV = () => {
    const headers = ['Case ID', 'Params', 'Final Residual', 'Execution Time (s)', 'Diff %'];
    const csvRows = sorted.map(r => [
      r.case_id,
      r.params,
      r.final_residual?.toExponential(4) ?? '',
      r.execution_time?.toFixed(1) ?? '',
      r.diff_pct !== null ? r.diff_pct.toFixed(2) + '%' : '',
    ]);
    const csv = [headers, ...csvRows].map(r => r.map(c => `"${c}"`).join(',')).join('\n');
    const blob = new Blob([csv], { type: 'text/csv' });
    const a = document.createElement('a');
    a.href = URL.createObjectURL(blob);
    const ts = new Date().toISOString().replace(/[:.]/g, '-').slice(0, 19);
    a.download = `comparison_metrics_${ts}.csv`;
    a.click();
    URL.revokeObjectURL(a.href);
  };

  if (rows.length === 0) {
    return (
      <div style={{ textAlign: 'center', padding: '48px', color: '#666' }}>
        No metrics available.
      </div>
    );
  }

  const SortIndicator = ({ col }: { col: SortKey }) => (
    sortKey === col ? (
      <span style={{ marginLeft: '4px' }}>{sortAsc ? '▲' : '▼'}</span>
    ) : null
  );

  return (
    <div style={{ padding: '16px' }}>
      <div style={{ overflowX: 'auto' }}>
        <table style={{ width: '100%', borderCollapse: 'collapse', fontSize: '14px' }}>
          <thead>
            <tr style={{ background: '#f5f5f7', borderBottom: '2px solid #ddd' }}>
              {(['case_id', 'params', 'final_residual', 'execution_time', 'diff_pct'] as SortKey[]).map(col => (
                <th
                  key={col}
                  onClick={() => handleSort(col)}
                  style={{
                    padding: '10px 12px',
                    textAlign: 'left',
                    cursor: 'pointer',
                    fontWeight: 600,
                    fontSize: '12px',
                    textTransform: 'capitalize',
                    userSelect: 'none',
                  }}
                >
                  {col.replace(/_/g, ' ')}
                  <SortIndicator col={col} />
                </th>
              ))}
            </tr>
          </thead>
          <tbody>
            {sorted.map(row => (
              <tr
                key={row.case_id}
                style={{ borderBottom: '1px solid #eee' }}
              >
                <td style={{ padding: '8px 12px', fontFamily: 'monospace', fontSize: '12px' }}>
                  {row.case_id}
                </td>
                <td style={{ padding: '8px 12px', fontFamily: 'monospace', fontSize: '12px' }}>
                  {row.params}
                </td>
                <td style={{ padding: '8px 12px', fontFamily: 'monospace' }}>
                  {row.final_residual != null ? row.final_residual.toExponential(4) : '—'}
                </td>
                <td style={{ padding: '8px 12px', fontFamily: 'monospace' }}>
                  {row.execution_time != null ? row.execution_time.toFixed(1) + 's' : '—'}
                </td>
                <td style={{
                  padding: '8px 12px',
```

```tsx
/**
 * ComparisonPage — Cross-case comparison UI (PIPE-12).
 *
 * Route: /comparisons/new
 *
 * Two-panel layout:
 *  - Left (320px fixed): Case selector — multi-select with provenance metadata
 *  - Right: ComparisonView with 3 tabs
 *
 * Pattern from 34-UI-SPEC.md + SweepDetailPage.tsx.
 */

import React, { useState, useEffect, useCallback } from 'react';
import { useNavigate, useSearchParams } from 'react-router-dom';
import { api } from '../services/api';
import type {
  SweepCase,
  ComparisonResponse,
  ProvenanceMismatchItem,
} from '../services/types';
import ConvergenceOverlay from '../components/ConvergenceOverlay';
import DeltaFieldViewer from '../components/DeltaFieldViewer';
import MetricsTable from '../components/MetricsTable';

type TabId = 'convergence' | 'delta' | 'metrics';

// Case series colors (up to 8 — from 34-UI-SPEC.md)
const SERIES_COLORS = [
  '#ef4444', '#22c55e', '#3b82f6', '#a855f7',
  '#f59e0b', '#06b6d4', '#ec4899', '#84cc16',
];

// Provenance warning banner
function ProvenanceWarningBanner({
  mismatches,
  onDismiss,
}: {
  mismatches: ProvenanceMismatchItem[];
  onDismiss: () => void;
}) {
  if (!mismatches.length) return null;
  return (
    <div style={{
      display: 'flex',
      alignItems: 'flex-start',
      gap: '12px',
      padding: '12px 16px',
      background: 'rgba(245, 158, 11, 0.1)',
      borderLeft: '4px solid #f59e0b',
      marginBottom: '16px',
    }}>
      <svg width="16" height="16" viewBox="0 0 24 24" fill="#f59e0b" style={{ flexShrink: 0, marginTop: '2px' }}>
        <path d="M12 2L2 22h20L12 2zm0 15v2m0 4h.01" stroke="#f59e0b" strokeWidth="2" fill="none" />
      </svg>
      <div style={{ flex: 1 }}>
        <strong style={{ color: '#92400e' }}>Provenance mismatch detected.</strong>
        <div style={{ fontSize: '12px', color: '#78350f', marginTop: '4px' }}>
          {mismatches.map(m => (
            <span key={m.field}> {m.field}: [{m.values.join(', ')}];</span>
          ))}
          {' '}Comparison results may not be meaningful.
        </div>
      </div>
      <button
        onClick={onDismiss}
        style={{ background: 'none', border: 'none', cursor: 'pointer', color: '#92400e', fontSize: '16px' }}
      >
        ×
      </button>
    </div>
  );
}

// Case selector item
function CaseSelectorItem({
  caseItem,
  selected,
  color,
  provenanceMismatch,
  onToggle,
}: {
  caseItem: SweepCase;
  selected: boolean;
  color: string;
  provenanceMismatch: boolean;
  onToggle: () => void;
}) {
  const paramStr = Object.entries(caseItem.param_combination || {})
    .map(([k, v]) => `${k}=${v}`)
    .join(', ');

  return (
    <label
      style={{
        display: 'flex',
        alignItems: 'flex-start',
        gap: '8px',
        padding: '8px',
        cursor: 'pointer',
        borderLeft: selected ? `3px solid ${color}` : '3px solid transparent',
        background: selected ? 'rgba(233, 69, 96, 0.05)' : 'transparent',
        transition: 'background 0.15s',
      }}
    >
      <input
        type="checkbox"
        checked={selected}
        onChange={onToggle}
        style={{ marginTop: '2px', flexShrink: 0 }}
      />
      <div style={{ flex: 1, minWidth: 0 }}>
        <div style={{ display: 'flex', justifyContent: 'space-between', alignItems: 'center' }}>
          <span style={{ fontFamily: 'monospace', fontSize: '12px' }}>{caseItem.id}</span>
          <span style={{
            fontSize: '10px',
            padding: '1px 4px',
            background: '#f5f5f7',
            borderRadius: '2px',
            fontFamily: 'monospace',
          }}>
            {caseItem.combination_hash}
          </span>
        </div>
        <div style={{ fontSize: '12px', color: '#666', marginTop: '2px' }}>{paramStr}</div>
        {/* Provenance chips */}
        <div style={{ display: 'flex', gap: '4px', marginTop: '4px', flexWrap: 'wrap' }}>
          {caseItem.openfoam_version && (
            <span style={{ fontSize: '10px', padding: '1px 4px', background: '#e8f0fe', borderRadius: '2px' }}>
              OF:{caseItem.openfoam_version}
            </span>
          )}
          {caseItem.compiler_version && (
            <span style={{ fontSize: '10px', padding: '1px 4px', background: '#e8f0fe', borderRadius: '2px' }}>
              GCC:{caseItem.compiler_version}
            </span>
          )}
          {caseItem.mesh_seed_hash && (
            <span style={{ fontSize: '10px', padding: '1px 4px', background: '#e8f0fe', borderRadius: '2px' }}>
              mesh:{caseItem.mesh_seed_hash.slice(0, 6)}
            </span>
          )}
        </div>
        {provenanceMismatch && (
          <div style={{
            fontSize: '10px',
            padding: '2px 6px',
            background: 'rgba(245, 158, 11, 0.15)',
            border: '1px solid #f59e0b',
            borderRadius: '2px',
            color: '#92400e',
            marginTop: '4px',
            display: 'inline-block',
          }}>
            Provenance mismatch
          </div>
        )}
      </div>
    </label>
  );
}

export default function ComparisonPage() {
  const navigate = useNavigate();
  const [searchParams] = useSearchParams();
  const existingComparisonId = searchParams.get('id');  // load existing comparison

  // State
  const [allCases, setAllCases] = useState<SweepCase[]>([]);
  const [selectedIds, setSelectedIds] = useState<Set<string>>(new Set());
  const [activeTab, setActiveTab] = useState<TabId>('convergence');
  const [comparison, setComparison] = useState<ComparisonResponse | null>(null);
  const [loading, setLoading] = useState(false);
  const [error, setError] = useState<string | null>(null);
  const [warningDismissed, setWarningDismissed] = useState(false);

  // Load all completed cases on mount
  useEffect(() => {
    api.getCases().then(cases => {
      setAllCases(cases);
    }).catch(console.error);
  }, []);

  // Load existing comparison if ID provided
  useEffect(() => {
    if (existingComparisonId) {
      api.getComparison(existingComparisonId).then(setComparison).catch(console.error);
    }
  }, [existingComparisonId]);

  const toggleCase = useCallback((id: string) => {
    setSelectedIds(prev => {
      const next = new Set(prev);
      if (next.has(id)) next.delete(id);
      else next.add(id);
      return next;
    });
    // Clear comparison when selection changes
    setComparison(null);
  }, []);

  const runComparison = useCallback(async () => {
    if (selectedIds.size < 2) return;
    const ids = Array.from(selectedIds);
    const referenceId = ids[0];  // first selected as reference
    setLoading(true);
    setError(null);
    try {
      const result = await api.createComparison({
        reference_case_id: referenceId,
        case_ids: ids,
        delta_case_a_id: ids[0],
        delta_case_b_id: ids[1],
        delta_field_name: 'p',
      });
      setComparison(result);
    } catch (e: any) {
      setError(`Comparison failed: ${e.message}`);
    } finally {
      setLoading(false);
    }
  }, [selectedIds]);

  const selectedCases = allCases.filter(c => selectedIds.has(c.id));

  // Build convergence series for overlay
  const convergenceSeries = comparison
    ? Object.entries(comparison.convergence_data || {}).map(([caseId, data], i) => {
        const caseItem = allCases.find(c => c.id === caseId);
        const params = caseItem
          ? Object.entries(caseItem.param_combination || {}).map(([k, v]) => `${k}=${v}`).join(', ')
          : caseId;
        return {
          caseId,
          label: `${caseId} (${params})`,
          color: SERIES_COLORS[i % SERIES_COLORS.length],
          data,
        };
      })
    : [];

  // Provenance mismatch detection (compare openfoam_version across selected)
  const provenanceMismatch = comparison?.provenance_mismatch ?? [];

  // Check mesh hash mismatch for delta field warning
  const meshMismatch = selectedCases.some(c => {
    const first = selectedCases[0];
    return first && c.mesh_seed_hash && first.mesh_seed_hash &&
      c.mesh_seed_hash !== first.mesh_seed_hash;
  });

  // Tab bar
  const tabs: { id: TabId; label: string }[] = [
    { id: 'convergence', label: 'Convergence' },
    { id: 'delta', label: 'Delta Field' },
    { id: 'metrics', label: 'Metrics' },
  ];

  return (
    <div style={{ display: 'flex', height: 'calc(100vh - 60px)' }}>
      {/* LEFT: Case Selector Panel */}
      <div style={{
        width: '320px',
        flexShrink: 0,
        borderRight: '1px solid #ddd',
        overflowY: 'auto',
        background: '#fafafa',
        display: 'flex',
        flexDirection: 'column',
      }}>
        <div style={{ padding: '16px', borderBottom: '1px solid #ddd' }}>
          <h3 style={{ margin: 0, fontSize: '16px', fontWeight: 600 }}>Select Cases to Compare</h3>
          <p style={{ margin: '4px 0 0', fontSize: '12px', color: '#666' }}>
            {selectedIds.size} selected
          </p>
        </div>

        {allCases.length === 0 ? (
          <div style={{ padding: '32px 16px', textAlign: 'center', color: '#666' }}>
            <div style={{ fontSize: '14px', marginBottom: '8px' }}>No Completed Cases</div>
            <div style={{ fontSize: '12px' }}>
              Run a parametric sweep to generate cases, then return here to compare results.
            </div>
          </div>
        ) : (
          <div style={{ flex: 1, overflowY: 'auto', padding: '8px 0' }}>
            {allCases.map((caseItem, i) => (
              <CaseSelectorItem
                key={caseItem.id}
                caseItem={caseItem}
                selected={selectedIds.has(caseItem.id)}
                color={SERIES_COLORS[i % SERIES_COLORS.length]}
                provenanceMismatch={
                  provenanceMismatch.some(m => m.field === 'openfoam_version' && m.values.includes(caseItem.openfoam_version || ''))
                }
                onToggle={() => toggleCase(caseItem.id)}
              />
            ))}
          </div>
        )}

        <div style={{ padding: '16px', borderTop: '1px solid #ddd' }}>
          <button
            onClick={runComparison}
            disabled={selectedIds.size < 2 || loading}
            title={selectedIds.size < 2 ? 'Select at least 2 cases' : undefined}
            style={{
              width: '100%',
              padding: '10px 16px',
              background: selectedIds.size >= 2 ? '#e94560' : '#ccc',
              color: '#fff',
              border: 'none',
              borderRadius: '4px',
              cursor: selectedIds.size >= 2 ? 'pointer' : 'not-allowed',
              fontWeight: 600,
            }}
          >
            {loading ? 'Comparing...' : `Compare ${selectedIds.size} Cases`}
          </button>
          {error && (
            <div style={{ color: '#ef4444', fontSize: '12px', marginTop: '8px' }}>{error}</div>
          )}
        </div>
      </div>

      {/* RIGHT: ComparisonView */}
      <div style={{ flex: 1, display: 'flex', flexDirection: 'column', overflow: 'hidden' }}>
        {/* Tab bar */}
        <div style={{
          display: 'flex',
          borderBottom: '1px solid #ddd',
          height: '48px',
          background: '#fff',
        }}>
          {tabs.map(tab => (
            <button
              key={tab.id}
              onClick={() => setActiveTab(tab.id)}
              style={{
                padding: '0 24px',
                height: '48px',
                border: 'none',
                borderBottom: activeTab === tab.id ? '2px solid #e94560' : '2px solid transparent',
                background: 'none',
                cursor: 'pointer',
                fontWeight: activeTab === tab.id ? 600 : 400,
                color: activeTab === tab.id ? '#e94560' : '#666',
                fontSize: '14px',
              }}
            >
              {tab.label}
            </button>
          ))}
        </div>

        {/* Tab content */}
        <div style={{ flex: 1, overflow: 'auto', background: '#fff' }}>
          {!comparison ? (
            <div style={{
              display: 'flex',
              alignItems: 'center',
              justifyContent: 'center',
              height: '300px',
              color: '#666',
            }}>
              Select cases and click "Compare" to see results.
            </div>
          ) : (
            <>
              {activeTab === 'convergence' && (
                <div style={{ padding: '24px' }}>
                  {!warningDismissed && provenanceMismatch.length > 0 && (
                    <ProvenanceWarningBanner
                      mismatches={provenanceMismatch}
                      onDismiss={() => setWarningDismissed(true)}
                    />
                  )}
                  <ConvergenceOverlay
                    series={convergenceSeries}
                    referenceCaseId={comparison.reference_case_id}
                  />
                </div>
              )}

              {activeTab === 'delta' && comparison && (
                <div style={{ padding: '24px' }}>
                  {!warningDismissed && provenanceMismatch.length > 0 && (
                    <ProvenanceWarningBanner
                      mismatches={provenanceMismatch}
                      onDismiss={() => setWarningDismissed(true)}
                    />
                  )}
                  <DeltaFieldViewer
                    comparisonId={comparison.id}
                    deltaVtuUrl={comparison.delta_vtu_url}
                    deltaFieldName={comparison.delta_field_name || 'p'}
                    caseAId={comparison.delta_case_a_id || ''}
                    caseBId={comparison.delta_case_b_id || ''}
                    meshHashMismatch={meshMismatch}
                  />
                </div>
              )}

              {activeTab === 'metrics' && comparison && (
                <MetricsTable
                  rows={comparison.metrics_table}
                  referenceCaseId={comparison.reference_case_id}
                />
              )}
            </>
          )}
        </div>
      </div>
    </div>
  );
}
```
