**Step A — Camera state tracking and polling**

Add at module level or inside `main()`:

```python
import json

# Track last camera state to avoid duplicate posts
_last_camera_state = None

def _get_camera_state():
    """Read current camera position/focal point from ParaView view."""
    try:
        view = simple.GetActiveView()
        if view is None:
            return None
        camera = view.GetCamera()
        if camera is None:
            return None
        pos = camera.GetPosition()
        focal = camera.GetFocalPoint()
        return {
            "position": [float(pos[0]), float(pos[1]), float(pos[2])],
            "focalPoint": [float(focal[0]), float(focal[1]), float(focal[2])],
        }
    except Exception:
        return None

def _post_camera_if_changed(server_instance):
    """Post camera state to parent if it changed. Called by interval timer."""
    global _last_camera_state
    state = _get_camera_state()
    if state is None:
        return
    state_key = (tuple(state["position"]), tuple(state["focalPoint"]))
    last_key = (
        tuple(_last_camera_state["position"]) if _last_camera_state else None,
        tuple(_last_camera_state["focalPoint"]) if _last_camera_state else None,
    )
    if state_key != last_key:
        _last_camera_state = state
        # Schedule post to parent via JavaScript
        server_instance.js_call(
            "window._trameBridge && window._trameBridge.postMessage("
            + json.dumps({"type": "camera", **state})
            + ", '*')"
        )
```

**Step B — postMessage handler (JavaScript injected into the page)**

Add to `main()` before `server.start()`:

```python
server.add_custom_script("""
// Inject postMessage listener and bridge into the trame page
window._trameBridge = window._trameBridge || {};

// Listen for postMessages from the React parent
window.addEventListener('message', function(event) {
    var data = event.data;
    if (!data || !data.type) return;

    var view = window._trameView;
    if (!view) return;

    // Route messages to trame server state setters or ctrl methods
    switch (data.type) {
        case 'field':
            if (window._trameServer && window._trameServer.state) {
                window._trameServer.state.field = data.field;
            }
            break;
        case 'slice':
            if (window._trameServer && window._trameServer.state) {
                window._trameServer.state.slice_axis = data.axis;
                window._trameServer.state.slice_origin = data.origin;
            }
            break;
        case 'slice_off':
            if (window._trameServer && window._trameServer.state) {
                window._trameServer.state.slice_axis = null;
            }
            break;
        case 'color_preset':
            if (window._trameServer && window._trameServer.state) {
                window._trameServer.state.color_preset = data.preset;
            }
            break;
        case 'scalar_range':
            if (window._trameServer && window._trameServer.state) {
                window._trameServer.state.scalar_range_mode = data.mode;
                window._trameServer.state.scalar_range_min = data.min;
                window._trameServer.state.scalar_range_max = data.max;
            }
            break;
        case 'volume_toggle':
            if (window._trameServer && window._trameServer.ctrl) {
                window._trameServer.ctrl.on_volume_rendering_toggle(data.field_name || '', data.enabled);
            }
            break;
        case 'timestep':
            if (window._trameServer && window._trameServer.state) {
                window._trameServer.state.time_step_index = data.index;
            }
            break;
        case 'clip_create':
            if (window._trameServer && window._trameServer.ctrl) {
                window._trameServer.ctrl.on_filter_clip_create(data.insideOut, data.scalarValue);
            }
            break;
        case 'contour_create':
            if (window._trameServer && window._trameServer.ctrl) {
                window._trameServer.ctrl.on_filter_contour_create(data.isovalues);
            }
            break;
        case 'streamtracer_create':
            if (window._trameServer && window._trameServer.ctrl) {
                window._trameServer.ctrl.on_filter_streamtracer_create(data.direction, data.maxSteps);
            }
            break;
        case 'filter_delete':
            if (window._trameServer && window._trameServer.ctrl) {
                window._trameServer.ctrl.on_filter_delete(data.filterId);
            }
            break;
        case 'filter_list':
            if (window._trameServer && window._trameServer.state) {
                window._trameServer.state.filter_list_request = True;
            }
            break;
        case 'volume_status':
            if (window._trameServer && window._trameServer.state) {
                window._trameServer.state.volume_rendering_status_request = True;
            }
            break;
        case 'screenshot':
            window._trameServer && window._trameServer.ctrl && window._trameServer.ctrl.on_screenshot(data.width, data.height);
            break;
    }
});

// Make view available to bridge
window._trameView = view;

// Start camera polling every 500ms
setInterval(function() {
    if (window._trameServer) {
        // Use server state to trigger camera read on Python side
        window._trameServer.state.camera_poll_trigger = (window._trameServer.state.camera_poll_trigger || 0) + 1;
    }
}, 500);
""")
```

**Step C — Camera poll state handler**

Add `@state.change("camera_poll_trigger")` handler (fires when the polling counter increments):

```python
@state.change("camera_poll_trigger")
def on_camera_poll(camera_poll_trigger, **kwargs):
    """Fires every ~500ms when camera_poll_trigger increments.
    Posts camera state to parent via js_call."""
    global _last_camera_state
    state = _get_camera_state()
    if state is None:
        return
    state_key = (tuple(state["position"]), tuple(state["focalPoint"]))
    last_key = (
        tuple(_last_camera_state["position"]) if _last_camera_state else None,
        tuple(_last_camera_state["focalPoint"]) if _last_camera_state else None,
    )
    if state_key != last_key:
        _last_camera_state = state
        # Use js_call to execute JS in browser
        server.js_call(
            "window._trameBridge && window._trameBridge.postMessage("
            + json.dumps({"type": "camera", **state})
            + ", '*')"
        )
```

Note: `server.js_call()` is the trame API for executing JavaScript in the client browser. If this API is not available in this trame version, use an alternative approach: store camera state in `state.camera_position` and `state.camera_focal_point`, and use a Vue component's `$nextTick` + `setInterval` to read from `window._trameServer.state` and post to parent.

**Step D — Screenshot handler**

Add `@ctrl.add` for screenshot:

```python
@ctrl.add
def on_screenshot(width: int, height: int):
    """Handle screenshot request from React bridge.
    Returns screenshot as base64 via postMessage."""
    try:
        view = simple.GetRenderView()
        if view is None:
            return
        # Set viewport size before rendering
        view.ViewSize = [width, height]
        simple.Render()
        # Capture using trame's html_view
        if hasattr(server, 'html_view'):
            img = server.html_view.screenshot(fill=False, quality=95)
            if img:
                # img is base64 string — post to parent
                server.js_call(
                    "window._trameBridge && window._trameBridge.postMessage("
                    + json.dumps({"type": "screenshot_data", "image": img})
                    + ", '*')"
                )
        return {"success": True}
    except Exception as e:
        return {"success": False, "error": str(e)}
```

Note: If `server.html_view` is not available, use ParaView's `SaveScreenshot` to a bytes buffer and encode base64. Post via `js_call`.

```python
@state.change("slice_axis", "slice_origin")
def on_slice_change(slice_axis, slice_origin, **kwargs):
    """When client sets slice axis and origin, create/update slice filter."""
    if slice_axis is None:
        # Remove slice — hide or delete slice filter
        try:
            slice_filter = getattr(state, '_slice_filter', None)
            if slice_filter:
                simple.Delete(slice_filter)
                state._slice_filter = None
            simple.Render()
            ctrl.view_update()
        except Exception:
            pass
        return
    try:
        source = simple.GetActiveSource()
        if source is None:
            return
        # Reuse or create slice filter
        slice_filter = getattr(state, '_slice_filter', None)
        if slice_filter is None:
            slice_filter = simple.Slice(Input=source)
            state._slice_filter = slice_filter
        normal_map = {"X": [1, 0, 0], "Y": [0, 1, 0], "Z": [0, 0, 1]}
        slice_filter.SliceType = "Plane"
        slice_filter.SliceType.Normal = normal_map.get(slice_axis, [0, 0, 1])
        slice_filter.SliceType.Origin = slice_origin
        simple.Render()
        ctrl.view_update()
    except Exception:
        pass
```

```python
from trame.html import vue

class PostMessageBridge(vue.VueComponent):
    """Lightweight Vue component that postMessages to parent."""
    template = """
    <div ref="bridge" :style="{display:'none'}">
      <div v-for="(msg, i) in pendingMessages" :key="i">
        {{ postToParent(msg) }}
      </div>
    </div>
    """
    def __init__(self, server):
        super().__init__(server)
        self.state.pendingMessages = []
        self.server.state.change('camera_post_data')(self._on_camera)
        self.server.state.change('screenshot_post_data')(self._on_screenshot)

    def postToParent(self, msg):
        self.server.js_call(f"window.parent.postMessage({json.dumps(msg)}, '*')")

    @property
    def pendingMessages(self):
        return self.server.state.pendingMessages
```
