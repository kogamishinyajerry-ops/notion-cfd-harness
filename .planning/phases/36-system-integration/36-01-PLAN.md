```python
class MetricDelta(BaseModel):
    """Single metric comparison result."""
    metric_name: str = Field(..., description="Metric name (e.g., shock_angle_deg, drag_coefficient_cd)")
    computed_value: float = Field(..., description="Computed value from solver")
    reference_value: float = Field(..., description="Literature reference value")
    delta_pct: float = Field(..., description="Percentage deviation: (computed - reference) / reference * 100")
    tolerance_pct: float = Field(default=5.0, description="Acceptable tolerance percentage")
    status: str = Field(..., description="PASS if abs(delta_pct) <= tolerance_pct, FAIL otherwise")


class LiteratureComparison(BaseModel):
    """Result of comparing a computed CFD result against literature GoldStandard.

    Produced by ComparisonService.validate_against_gold_standard().
    """
    case_id: str = Field(..., description="GoldStandard case ID (e.g., SU2-02, OF-04)")
    platform: str = Field(..., description="Platform: OpenFOAM or SU2")
    reference_source: str = Field(..., description="Literature citation (e.g., Ghia 1982, NASA TN D-556)")
    metrics: List[MetricDelta] = Field(default_factory=list, description="Per-metric comparison results")
    overall_delta_pct: Optional[float] = Field(None, description="Average of abs(metrics[].delta_pct)")
    passed: bool = Field(..., description="True if all metrics within tolerance")

    def __init__(self, **data):
        # Auto-compute overall_delta_pct from metrics if not provided
        if 'metrics' in data and 'overall_delta_pct' not in data:
            deltas = [abs(m.delta_pct) for m in data['metrics'] if hasattr(m, 'delta_pct')]
            data['overall_delta_pct'] = sum(deltas) / len(deltas) if deltas else None
        super().__init__(**data)
```

```python
async def gold_standard_wrapper(step, cancel_event: threading.Event) -> StepResult:
    """
    GS-05: Execute a full GoldStandard case lifecycle:
      1. Generate case from GoldStandardRegistry spec_factory
      2. Run solver via JobService
      3. Monitor convergence
      4. Validate results against literature reference via ComparisonService

    Params required:
      - gold_standard_case_id: str — whitelist case ID (e.g., "SU2-02", "OF-04")
      - case_name: Optional[str] — override case name (default from registry)

    Returns:
      StepResult(status=SUCCESS) if all metrics pass literature validation.
      StepResult(status=VALIDATION_FAILED) if any metric exceeds tolerance.
      StepResult(status=ERROR) on unexpected exception or cancellation.
    """
    if cancel_event.is_set():
        return StepResult(
            status=StepResultStatus.ERROR, exit_code=1,
            diagnostics={"cancelled": True}
        )

    try:
        from api_server.services.comparison_service import get_comparison_service_for_validation
        from knowledge_compiler.phase1.gold_standards.registry import get_gold_standard_registry

        params = step.params
        case_id = params.get("gold_standard_case_id")
        if not case_id:
            return StepResult(
                status=StepResultStatus.ERROR, exit_code=1,
                diagnostics={"error": "gold_standard_case_id param required"}
            )

        registry = get_gold_standard_registry()
        comparison_svc = get_comparison_service_for_validation()

        # Phase 1: Generate case
        if cancel_event.is_set():
            return StepResult(status=StepResultStatus.ERROR, exit_code=1, diagnostics={"cancelled": True})

        case_dir = None
        try:
            spec = registry.get_spec(case_id)
            from knowledge_compiler.phase2.execution_layer.generic_case_generator import GenericOpenFOAMCaseGenerator
            generator_kwargs = spec.to_dict() if hasattr(spec, 'to_dict') else dict(spec)
            generator = GenericOpenFOAMCaseGenerator(**generator_kwargs)
            output = await asyncio.to_thread(generator.generate)
            case_dir = str(output.get("case_dir", output) if isinstance(output, dict) else output)
        except Exception as e:
            logger.warning(f"GoldStandard case generation failed for {case_id}: {e}")
            # Fallback: use case_id as case_dir
            case_dir = f"/tmp/gs_case_{case_id}"

        # Phase 2: Run solver
        if cancel_event.is_set():
            return StepResult(status=StepResultStatus.ERROR, exit_code=1, diagnostics={"cancelled": True})

        from api_server.services.job_service import JobService, _JOBS
        job_service = JobService()
        submission = JobSubmission(
            case_id=case_id,
            job_type="run",
            parameters={"case_dir": case_dir, "case_id": case_id, "pipeline_id": params.get("pipeline_id")},
            async_mode=True,
        )
        job = job_service.submit_job(submission)
        job_id = job.job_id

        # Poll for completion (reuse run_wrapper polling pattern)
        poll_interval = 5.0
        max_wait = 7200
        elapsed = 0
        job_completed = False

        while elapsed < max_wait:
            if cancel_event.is_set():
                return StepResult(status=StepResultStatus.ERROR, exit_code=1, diagnostics={"cancelled": True, "job_id": job_id})
            await asyncio.sleep(poll_interval)
            elapsed += poll_interval
            current_job = _JOBS.get(job_id)
            if current_job is None:
                break
            status_val = current_job.status.value if hasattr(current_job.status, 'value') else str(current_job.status)
            if status_val == "completed":
                job_completed = True
                break
            if status_val == "failed":
                return StepResult(
                    status=StepResultStatus.ERROR, exit_code=1,
                    diagnostics={"error": "Solver job failed", "job_id": job_id}
                )

        if not job_completed:
            return StepResult(
                status=StepResultStatus.ERROR, exit_code=1,
                diagnostics={"error": "GoldStandard solver timeout", "job_id": job_id}
            )

        # Phase 3: Build report_spec from job result and validate
        current_job = _JOBS.get(job_id)
        solver_result = current_job.result if current_job else {}

        # Build report_spec dict for validation
        report_spec = {
            "case_id": case_id,
            "case_dir": case_dir,
            "solver_result": solver_result,
        }

        # Phase 4: Validate against gold standard
        literature_comp = comparison_svc.validate_against_gold_standard(case_id, report_spec)

        diagnostics = {
            "job_id": job_id,
            "case_dir": case_dir,
            "literature_comparison": literature_comp.model_dump() if hasattr(literature_comp, 'model_dump') else dict(literature_comp),
            "metrics_passed": literature_comp.passed,
        }

        if literature_comp.passed:
            return StepResult(
                status=StepResultStatus.SUCCESS, exit_code=0,
                validation_checks={"literature_validation": True},
                diagnostics=diagnostics,
            )
        else:
            return StepResult(
                status=StepResultStatus.VALIDATION_FAILED, exit_code=0,
                validation_checks={"literature_validation": False},
                diagnostics=diagnostics,
            )

    except Exception as e:
        logger.exception(f"gold_standard_wrapper failed for step {step.step_id}")
        return StepResult(
            status=StepResultStatus.ERROR, exit_code=1,
            diagnostics={"exception": str(e)}
        )
```

```python
    def validate_against_gold_standard(
        self, case_id: str, report_spec: Dict[str, Any]
    ) -> LiteratureComparison:
        """
        GS-06: Validate a computed CFD result against literature GoldStandard.

        Loads reference data from GoldStandardRegistry for case_id, extracts
        scalar metrics from report_spec, compares per-metric, returns
        LiteratureComparison DTO with per-metric delta percentages.

        Args:
            case_id: GoldStandard case ID (e.g., "SU2-02", "OF-04")
            report_spec: Dict containing solver_result with computed metrics

        Returns:
            LiteratureComparison DTO with per-metric comparisons and overall pass/fail
        """
        from knowledge_compiler.phase1.gold_standards.registry import get_gold_standard_registry

        registry = get_gold_standard_registry()

        # Resolve module_case_id from whitelist_id if needed
        module_case_id = registry.get_module_case_id(case_id) or case_id
        whitelist_case = registry.get_whitelist_case(case_id)
        platform = whitelist_case.platform if whitelist_case else "Unknown"
        reference_source = whitelist_case.source_provenance if whitelist_case else ""

        # Load reference data from registry
        ref_data = registry.get_reference_data(module_case_id)
        if ref_data is None:
            # No reference data registered — return a passive comparison
            return LiteratureComparison(
                case_id=case_id,
                platform=platform,
                reference_source=reference_source or "No reference registered",
                metrics=[],
                passed=True,
            )

        # Extract computed metrics from report_spec
        solver_result = report_spec.get("solver_result", {})
        computed_metrics = {}

        # Known metric extraction patterns from solver_result
        if "shock_angle_deg" in ref_data:
            computed_metrics["shock_angle_deg"] = solver_result.get("shock_angle_deg")
        if "drag_coefficient_cd" in ref_data:
            computed_metrics["drag_coefficient_cd"] = solver_result.get("drag_coefficient_cd") or solver_result.get("cd")
        if "average_cf" in ref_data:
            computed_metrics["average_cf"] = solver_result.get("average_cf")
        if "strouhal_number" in ref_data:
            computed_metrics["strouhal_number"] = solver_result.get("strouhal_number")
        if "CL" in ref_data:
            computed_metrics["CL"] = solver_result.get("CL")
        if "column_height" in ref_data:
            computed_metrics["column_height"] = solver_result.get("column_height") or solver_result.get("H")
        if "centerline_velocity" in ref_data:
            computed_metrics["centerline_velocity"] = solver_result.get("centerline_velocity")
        # Generic fallback: any key in ref_data that's a float
        for key, ref_val in ref_data.items():
            if isinstance(ref_val, (int, float)) and key not in computed_metrics:
                computed_val = solver_result.get(key)
                if computed_val is not None:
                    computed_metrics[key] = computed_val

        # Build per-metric comparisons
        metric_deltas: List[MetricDelta] = []
        all_pass = True
        tolerance_pct = 5.0  # default 5% tolerance

        for metric_name, reference_value in ref_data.items():
            if not isinstance(reference_value, (int, float)):
                continue

            computed_value = computed_metrics.get(metric_name)
            if computed_value is None:
                # Try alternate names
                if metric_name == "CL":
                    computed_value = solver_result.get("CL")
                elif metric_name == "CD":
                    computed_value = solver_result.get("CD") or solver_result.get("drag_coefficient_cd")

            if computed_value is None:
                # Metric not in result — skip with warning
                logger.warning(f"Computed value for {metric_name} not found in solver_result")
                continue

            delta_pct = ((computed_value - reference_value) / abs(reference_value)) * 100
            status = "PASS" if abs(delta_pct) <= tolerance_pct else "FAIL"
            if status == "FAIL":
                all_pass = False

            metric_deltas.append(MetricDelta(
                metric_name=metric_name,
                computed_value=float(computed_value),
                reference_value=float(reference_value),
                delta_pct=round(delta_pct, 4),
                tolerance_pct=tolerance_pct,
                status=status,
            ))

        overall_delta = None
        if metric_deltas:
            overall_delta = round(sum(abs(m.delta_pct) for m in metric_deltas) / len(metric_deltas), 4)

        return LiteratureComparison(
            case_id=case_id,
            platform=platform,
            reference_source=reference_source,
            metrics=metric_deltas,
            overall_delta_pct=overall_delta,
            passed=all_pass,
        )
```
