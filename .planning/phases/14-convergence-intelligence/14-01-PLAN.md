"""
Divergence detection for CFD simulation residual monitoring.

Uses rolling 5-iteration window per variable to detect when residuals
increase 5 consecutive times, indicating solver divergence.
"""

import logging
from collections import deque
from typing import Any, Callable, Dict, Awaitable, Optional

logger = logging.getLogger(__name__)


class DivergenceDetector:
    """
    Tracks residual history per variable with rolling window.

    Fires alert when residual for any variable (Ux, Uy, Uz, p) increases
    for 5 consecutive iterations after iteration >= 5.

    Per locked decision:
    - Rolling 5-iteration window per variable
    - Alert fires when residual increases 5 consecutive times
    - Only armed after iteration >= 5 (avoiding startup spikes)
    - No re-alert once divergence state is set per variable
    """

    VARIABLES = ("Ux", "Uy", "Uz", "p")

    def __init__(
        self,
        callback: Callable[[Dict[str, Any]], Awaitable[None]],
        window_size: int = 5,
        arm_iteration: int = 5,
    ):
        """
        Initialize DivergenceDetector.

        Args:
            callback: Async callback to invoke with residual data or divergence alert.
            window_size: Number of iterations to track per variable (default 5).
            arm_iteration: Iteration number when detection becomes active (default 5).
        """
        self._callback = callback
        self._window_size = window_size
        self._arm_iteration = arm_iteration
        # Rolling window per variable: var_name -> deque of (iteration, residual_value)
        self._history: Dict[str, deque] = {
            var: deque(maxlen=window_size) for var in self.VARIABLES
        }
        # Track which variables have already diverged (no re-alert)
        self._diverged: Dict[str, bool] = {var: False for var in self.VARIABLES}

    async def __call__(self, residual_data: Dict[str, Any]) -> None:
        """
        Process residual update, detect divergence, forward to callback.

        Args:
            residual_data: Dict containing iteration, residuals, status, etc.
        """
        iteration = residual_data.get("iteration", 0)
        residuals = residual_data.get("residuals", {})
        status = residual_data.get("status", "running")

        # If job has ended (converged/diverged/stalled), reset divergence state
        if status in ("converged", "diverged", "stalled"):
            self._reset_divergence_state()

        # Check divergence for each variable
        alert_triggered = False
        diverged_var: Optional[str] = None

        for var in self.VARIABLES:
            value = residuals.get(var)
            if value is None:
                continue

            history = self._history[var]
            history.append((iteration, value))

            # Only check after arm_iteration and if not already diverged
            if iteration >= self._arm_iteration and not self._diverged[var]:
                if self._check_strictly_increasing(history):
                    self._diverged[var] = True
                    alert_triggered = True
                    diverged_var = var
                    break  # One alert per update, not one per variable

        # Always forward the original residual data
        await self._callback(residual_data)

        # If divergence detected, send alert
        if alert_triggered and diverged_var:
            history = self._history[diverged_var]
            previous_values = [v for _, v in history]
            await self._callback({
                "type": "divergence_alert",
                "job_id": residual_data.get("job_id"),
                "iteration": iteration,
                "diverged_variable": diverged_var,
                "current_value": residuals.get(diverged_var),
                "previous_values": previous_values,
                "message": f"Residual for {diverged_var} increased 5 consecutive iterations",
                "timestamp": residual_data.get("timestamp"),
            })
            logger.warning(
                f"Divergence detected: {diverged_var} at iteration {iteration}, "
                f"values: {previous_values}"
            )

    def _check_strictly_increasing(self, history: deque) -> bool:
        """
        Return True if deque has window_size entries and all are strictly increasing.

        Args:
            history: Deque of (iteration, residual_value) tuples.

        Returns:
            True if all values in window are strictly increasing.
        """
        if len(history) < self._window_size:
            return False
        values = [v for _, v in history]
        return all(values[i] < values[i + 1] for i in range(len(values) - 1))

    def _reset_divergence_state(self) -> None:
        """Reset divergence tracking for all variables."""
        for var in self.VARIABLES:
            self._diverged[var] = False
            self._history[var].clear()
