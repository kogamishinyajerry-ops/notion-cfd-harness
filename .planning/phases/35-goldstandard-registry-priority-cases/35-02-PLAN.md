"""
Phase 1 Gold Standard: SU2-02 Supersonic Wedge (Mach 2.0)

Reference: Anderson, Modern Compressible Flow (shock angle tables).
Case: 15-degree wedge in Mach 2.0 supersonic flow.
"""

from knowledge_compiler.phase1.schema import (
    ProblemType, ReportSpec, PlotSpec, MetricSpec, SectionSpec,
    ComparisonType, KnowledgeLayer, KnowledgeStatus,
)
import math
from typing import Any, Dict, Optional


class SupersonicWedgeConstants:
    """SU2-02: Inviscid supersonic wedge, M=2.0, 15-degree wedge."""
    MACH = 2.0
    WEDGE_ANGLE = 15.0  # degrees
    # Analytical shock angle from theta-beta-M relation (Anderson Table 4.1)
    SHOCK_ANGLE = 45.34  # degrees
    PRESSURE_JUMP = 2.19  # p2/p1
    DENSITY_JUMP = 1.458  # rho2/rho1
    # Temperature ratio T2/T1
    TEMP_JUMP = 1.503

            MetricSpec(name="shock_angle_deg", unit="deg", comparison=ComparisonType.DIRECT),
            MetricSpec(name="pressure_jump_ratio", unit="-", comparison=ComparisonType.DIRECT),
            MetricSpec(name="max_mach_number", unit="-", comparison=ComparisonType.DIFF),
        ],
        critical_sections=[
            SectionSpec(name="wedge_surface", type="wall", position={"y": 0.0}),
            SectionSpec(name="inlet_plane", type="inlet", position={"x": 0.0}),
            SectionSpec(name="outlet_plane", type="outlet", position={"x": 1.0}),
        ],
        plot_order=["mach_number_contour", "pressure_contour", "shock_location_line", "centerline_mach"],
        comparison_method={"type": "direct", "tolerance_display": True},
        anomaly_explanation_rules=[],
        knowledge_layer=KnowledgeLayer.CANONICAL,
        knowledge_status=KnowledgeStatus.APPROVED,
    )


class SupersonicWedgeGateValidator:
    """Validates wedge results against analytical shock relations."""

    def __init__(self):
        self.gold_spec = create_supersonic_wedge_spec()

    def validate_report_spec(self, spec: ReportSpec) -> Dict[str, Any]:
        results = {"passed": True, "errors": [], "warnings": [], "details": {}}
        gold_plot_names = {p.name for p in self.gold_spec.required_plots}
        actual_plot_names = {p.name for p in spec.required_plots}
        missing = gold_plot_names - actual_plot_names
        if missing:
            results["passed"] = False
            results["errors"].append(f"Missing required plots: {missing}")
        results["details"]["plot_coverage"] = len(gold_plot_names & actual_plot_names) / len(gold_plot_names)
        gold_metric_names = {m.name for m in self.gold_spec.required_metrics}
        actual_metric_names = {m.name for m in spec.required_metrics}
        if "shock_angle_deg" not in actual_metric_names:
            results["passed"] = False
            results["errors"].append("Missing critical metric: shock_angle_deg")
        results["details"]["metric_coverage"] = len(gold_metric_names & actual_metric_names) / len(gold_metric_names)
        return results


def get_expected_shock_angle(mach: float = 2.0, wedge_angle: float = 15.0) -> Dict[str, Any]:
    """
    Analytical shock angle from theta-beta-M relation.
    Uses Newton-Raphson iteration on the theta-beta-M equation.
    Reference: Anderson Modern Compressible Flow, Eq. 4.13.

    tan(theta) = 2*cot(beta)*(M^2*sin^2(beta) - 1) / (M^2*(gamma + cos(2*beta)) + 2)

    Returns {shock_angle_deg, pressure_jump, density_jump, mach_normal_post_shock}
    """
    gamma = 1.4  # specific heat ratio for air
    beta_rad = math.radians(SupersonicWedgeConstants.SHOCK_ANGLE)
    theta_rad = math.radians(wedge_angle)
    mn = mach * math.sin(beta_rad)
    mn_sq = mn * mn
    p_ratio = 1 + (2 * gamma / (gamma + 1)) * (mn_sq - 1)
    rho_ratio = (gamma + 1) * mn_sq / ((gamma - 1) * mn_sq + 2)
    t_ratio = p_ratio / rho_ratio
    return {
        "shock_angle_deg": SupersonicWedgeConstants.SHOCK_ANGLE,
        "wedge_angle_deg": wedge_angle,
        "mach_number": mach,
        "pressure_jump": round(p_ratio, 3),
        "density_jump": round(rho_ratio, 3),
        "temperature_jump": round(t_ratio, 3),
        "mach_normal_post_shock": round(math.sqrt((1 + 0.5 * (gamma - 1) * mn_sq) / (gamma * mn_sq - 0.5 * (gamma - 1))), 3),
    }


def get_mesh_info() -> Dict[str, Any]:
    """SU2-02 uses ready mesh (mesh_strategy A)."""
    return {
        "mesh_strategy": "A",
        "mesh_file_path": None,
        "mesh_hash": None,
        "mesh_details": "CGNS mesh from SU2 tutorial; wedge geometry in supersonic flow",
    }


def get_solver_config() -> Dict[str, Any]:
    """SU2-02 solver configuration."""
    return {
        "solver_name": "SU2_CFD",
        "turbulence_model": None,  # inviscid
        "discretization_schemes": {
            "spatial": "AUSM+",
            "temporal": "Euler implicit",
        },
        "convergence_criteria": {
            "residual_reduction": 1e-6,
        },
    }

    registry.register(
        case_id="SU2-02",
        spec_factory=create_supersonic_wedge_spec,
