```python
from dataclasses import dataclass
from enum import Enum

class GeometryType(Enum):
    SIMPLE_GRID = "simple_grid"       # Single rectangular block (cavity)
    BACKWARD_FACING_STEP = "backward_facing_step"  # Step geometry (BFS)
    BODY_IN_CHANNEL = "body_in_channel"  # Rectangular body in channel (cylinder)

class BCType(Enum):
    FIXED_VALUE = "fixedValue"
    ZERO_GRADIENT = "zeroGradient"
    SYMMETRY_PLANE = "symmetryPlane"
    WALL = "wall"
    EMPTY = "empty"
    PATCH = "patch"

class SolverType(Enum):
    ICO_FOAM = "icoFoam"
    SIMPLE_FOAM = "simpleFoam"
    PIMPLE_FOAM = "pimpleFoam"

@dataclass(frozen=True)
class BoundaryPatchSpec:
    name: str
    bc_type: BCType
    # For fixedValue: value string e.g. "(1.0 0 0)" for U or "0" for scalar
    value: str = ""

@dataclass(frozen=True)
class GeometrySpec:
    geometry_type: GeometryType
    x_min: float
    x_max: float
    y_min: float
    y_max: float
    thickness: float = 0.01
    # Body bounds (used only when geometry_type == BODY_IN_CHANNEL)
    body_x_min: float | None = None
    body_x_max: float | None = None
    body_y_min: float | None = None
    body_y_max: float | None = None

@dataclass(frozen=True)
class MeshSpec:
    nx: int = 40
    ny: int = 40
    # For BACKWARD_FACING_STEP: cell counts per section
    nx_inlet: int | None = None
    nx_outlet: int | None = None
    ny_lower: int | None = None
    ny_upper: int | None = None
    # For BODY_IN_CHANNEL: cell counts per sub-region
    nx_left: int | None = None
    nx_body: int | None = None
    nx_right: int | None = None
    ny_outer: int | None = None
    ny_body: int | None = None

@dataclass(frozen=True)
class PhysicsSpec:
    solver: SolverType
    reynolds_number: float
    # For laminar (icoFoam, pimpleFoam): nu is computed from Re and U
    # For RAS (simpleFoam): k, epsilon at inlet
    u_inlet: float = 1.0
    u_lid: float = 1.0           # For cavity
    k_inlet: float | None = None
    epsilon_inlet: float | None = None
    nu: float | None = None     # Override computed nu if provided
    end_time: float = 10.0
    delta_t: float = 0.001
    write_interval: float = 1.0
    max_co: float | None = None  # For pimpleFoam

@dataclass(frozen=True)
class BoundarySpec:
    # Maps patch name -> BoundaryPatchSpec
    patches: dict[str, BoundaryPatchSpec]

# Validation constants
MAX_CELL_COUNT: int = 1_000_000
MAX_DOMAIN_SIZE: float = 1000.0
MIN_DOMAIN_SIZE: float = 1e-6

def validate_geometry_spec(spec: GeometrySpec) -> list[str]:
    """Return list of validation error strings. Empty list = valid."""

def validate_mesh_spec(spec: MeshSpec, geometry: GeometrySpec) -> list[str]:
    """Return list of validation error strings. Empty list = valid."""

def validate_physics_spec(spec: PhysicsSpec, geometry: GeometrySpec) -> list[str]:
    """Return list of validation error strings. Empty list = valid."""
```
