## Protocol Context — ParaView Web Messages for PV-04

Based on ParaView Web protocol documentation and the existing message pattern in paraviewProtocol.ts:

**1. Camera Reset (PV-04.2):**
```json
{ "id": "pv-camera-reset", "method": "view.resetCamera", "params": {} }
```

**2. Slice Filter (PV-04.3) — Create slice on X/Y/Z axis:**
ParaView Web Slice filter uses `Slice.Create` method:
```json
{
  "id": "pv-slice",
  "method": "Slice.Create",
  "params": {
    "input": "{reader_id}",
    "normal": [1, 0, 0],
    "origin": [0, 0, 0]
  }
}
```
- `normal: [1,0,0]` = X plane, `[0,1,0]` = Y plane, `[0,0,1]` = Z plane
- `origin` is the point the plane passes through (array of 3 floats)

**3. Color Preset (PV-04.4) — Change lookup table:**
```json
{
  "id": "pv-lut",
  "method": "UpdateLUT",
  "params": {
    "property": "lookupTable",
    "preset": "Viridis"
  }
}
```
Valid presets: "Viridis", "BlueRed", "Grayscale"

**4. Scalar Range Mode (PV-04.6) — Auto vs Manual:**
```json
{
  "id": "pv-range",
  "method": "UpdateScalarRange",
  "params": {
    "mode": "auto"
  }
}
```
or manual:
```json
{
  "id": "pv-range",
  "method": "UpdateScalarRange",
  "params": {
    "mode": "manual",
    "min": 0.0,
    "max": 100.0
  }
}
```

**5. Scalar Bar / Legend (PV-04.5):**
```json
{
  "id": "pv-scalarbar",
  "method": "CreateScalarBar",
  "params": {
    "visible": true,
    "component": "ScalarBarWidget"
  }
}
```

```typescript
/**
 * Create a protocol message to reset the camera to default orientation.
 * Maps to PV-04.2: Camera reset button.
 */
export function createCameraResetMessage(): object {
  return {
    id: "pv-camera-reset",
    method: "view.resetCamera",
    params: {}
  };
}

/**
 * Create a protocol message to create or update an axis-aligned slice filter.
 * Maps to PV-04.3: Slice filter with adjustable origin.
 *
 * @param axis - "X" | "Y" | "Z" — the plane normal direction
 * @param origin - [number, number, number] — point the plane passes through
 */
export function createSliceMessage(axis: 'X' | 'Y' | 'Z', origin: [number, number, number]): object {
  const normalMap: Record<string, [number, number, number]> = {
    X: [1, 0, 0],
    Y: [0, 1, 0],
    Z: [0, 0, 1]
  };
  return {
    id: "pv-slice",
    method: "Slice.Create",
    params: {
      input: "OpenFOAMReader",  // The reader proxy name
      normal: normalMap[axis] ?? [0, 0, 1],
      origin: origin
    }
  };
}

/**
 * Create a protocol message to change the color lookup table preset.
 * Maps to PV-04.4: Color presets — Viridis, BlueRed, Grayscale.
 *
 * @param preset - "Viridis" | "BlueRed" | "Grayscale"
 */
export function createColorPresetMessage(preset: 'Viridis' | 'BlueRed' | 'Grayscale'): object {
  return {
    id: "pv-lut",
    method: "UpdateLUT",
    params: {
      property: "lookupTable",
      preset: preset
    }
  };
}

/**
 * Create a protocol message to set the scalar color range.
 * Maps to PV-04.6: Scalar range — Auto or Manual.
 *
 * @param mode - "auto" | "manual"
 * @param min - minimum value (used in manual mode)
 * @param max - maximum value (used in manual mode)
 */
export function createScalarRangeMessage(mode: 'auto' | 'manual', min?: number, max?: number): object {
  if (mode === 'auto') {
    return {
      id: "pv-range",
      method: "UpdateScalarRange",
      params: { mode: "auto" }
    };
  }
  return {
    id: "pv-range",
    method: "UpdateScalarRange",
    params: { mode: "manual", min: min ?? 0, max: max ?? 1 }
  };
}

/**
 * Create a protocol message to show or hide the scalar color bar (legend).
 * Maps to PV-04.5: Scalar bar with min/max values and labeled ticks.
 *
 * @param visible - true to show the scalar bar, false to hide
 */
export function createScalarBarMessage(visible: boolean): object {
  return {
    id: "pv-scalarbar",
    method: "CreateScalarBar",
    params: {
      visible: visible,
      component: "ScalarBarWidget"
    }
  };
}
```

These functions follow the same pattern as Phase 17 message builders (string IDs, object return type).

const renderSliceControls = () => {

    <div className="slice-controls">
      <div className="slice-row">

          {(['X', 'Y', 'Z'] as const).map((axis) => (

              onClick={() => {

                sendProtocolMessage(createSliceMessage(axis, sliceOrigin));
                sendProtocolMessage(createRenderMessage());

            >

          ))}
          <button
            className={`axis-btn ${sliceAxis === null ? 'active' : ''}`}
            onClick={() => {
              setSliceAxis(null);
              // ParaView Web: setting slice input to empty removes the filter
              // Send a render to update
              sendProtocolMessage(createRenderMessage());
            }}
          >
            Off
          </button>
        </div>
      </div>
      {sliceAxis !== null && (
        <div className="slice-origin-row">
          <label className="selector-label">Origin</label>
          <input
            type="range"
            className="origin-slider"
            min="-5"
            max="5"
            step="0.1"
            value={sliceOrigin[0]}
            onChange={(e) => {
              const val = parseFloat(e.target.value);
              const newOrigin: [number, number, number] = [val, 0, 0];
              setSliceOrigin(newOrigin);
              sendProtocolMessage(createSliceMessage(sliceAxis, newOrigin));
              sendProtocolMessage(createRenderMessage());
            }}
          />
          <span className="origin-value">{sliceOrigin[0].toFixed(1)}</span>
        </div>
      )}
    </div>
  );
};

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| client -> ParaView Web server (WebSocket) | Untrusted: user-supplied slice origin (float), preset name (string enum), scalar range values (float) cross this boundary |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-18-01 | Tamper | slice origin input | mitigate | `parseFloat()` converts input; out-of-range values are clamped by ParaView Web server internally. Origin slider constrained to -5..5 range. |
| T-18-02 | Tamper | scalar min/max inputs | mitigate | `parseFloat()` converts input; NaN values fall back to current state via `??` operator. Manual range inputs are type="number". |
| T-18-03 | Information disclosure | color preset, slice axis | accept | Display preferences are user intent only, no sensitive data. |
| T-18-04 | Denial of Service | rapid slider/input changes | mitigate | Each onChange fires a WebSocket message; ParaView Web server handles message rate. No client-side throttling needed for v1.4.0. |
</threat_model>
